
[Game Program File]
FUNCTION: "intro", PARAMS: none, RETURN TYPE: 1D multi-type array

    1D STRING ARRAY: "outcomes", LENGTH: 2, CONTENTS: ("That's a cool name!", "That's a weird name" )
    1D INT ARRAY: "waits", LENGTH: 2, CONTENTS: (50, 100)
    INT: "num", CONTENTS: random int (0, 2)

    STRING: "title", CONTENTS: Util.getMenu(), PARAMS: 2D STRING ARRAY: LENGTH: 1, CONTENTS: (("Adventure Game"))
    PRINTLINE title

    CALL Util.staggeredPrint(), PARAMS: "Why hello adventurer!"
    CALL Util.stagWait()

    CALL Util.staggeredPrint(), PARAMS: "What is your name?"
    PRINT "> "

    STRING: "name", CONTENTS: user input

    IF name equals ""

        PRINT error message
        SET name to "bob"

    CALL Util.staggeredPrint(), PARAMS: "{name}..."
    CALL Util.stagWait()

    CALL Util.staggeredPrint(), PARAMS: outcomes(INDEX: num), waits(INDEX: num)
    CALL Util.stagWait()

    STRING: "ageStr", CONTENTS: user input

    IF characters in ageStr

        PRINT error message
        INT: "age", CONTENTS: 12

    RETURN: 1D MULTI-TYPE ARRAY: CONTENTS: (name, age)



FUNCTION: "game1", PARAMS: 1D MULTI-TYPE ARRAY, RETURN TYPE: none / null

    STRING ARRAY: "riddles", LENGTH: 3, CONTENTS: ("What has 4 legs, but no eyes?",
                                                   "Why is 99 more than 100?",
                                                   "What do you call a liar on the phone?")




[Main Program File]
BOOL: "loop", CONTENTS: true

WHILE loop equals true

    1D MULTI-TYPE ARRAY: "playerData", LENGTH: 2, CONTENTS: ("", 0)

    SET playerData to Game.intro()

    CALL Game.changeGame()
    SET loop to Game.game1(), PARAMS: playerData

    CALL Game.changeGame()
    SET loop to Game.game2(), PARAMS: playerData

    CALL Game.changeGame()
    SET loop to Game.game3(), PARAMS: playerData

SET loop to Game.outro()
